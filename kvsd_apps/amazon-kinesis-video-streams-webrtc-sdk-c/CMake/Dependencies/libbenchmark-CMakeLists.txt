project(libbenchmark-download NONE)

include(ExternalProject)

  SOURCE_DIR        ${CMAKE_CURRENT_SOURCE_DIR}/../../CMake/Dependencies/benchmark

    -DBENCHMARK_ENABLE_GTEST_TESTS=OFF
  BUILD_ALWAYS    TRUE
  TEST_COMMAND    "")
