cmake_minimum_required(VERSION 2.8)

project(libgperftools-download NONE)

set(CONFIGURE_COMMAND ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libgperftools/configure --prefix=${OPEN_SRC_INSTALL_PREFIX})

include(ExternalProject)
ExternalProject_Add(project_libgperftools
    SOURCE_DIR        ${CMAKE_CURRENT_SOURCE_DIR}/../../CMake/Dependencies/gperftools-2.8
    DOWNLOAD_COMMAND  
                      cp -av ${CMAKE_CURRENT_SOURCE_DIR}/../../CMake/Dependencies/gperftools-2.8/. 
                            ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libgperftools
    PREFIX            ${CMAKE_CURRENT_BINARY_DIR}/build
    BINARY_DIR        ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libgperftools
    BUILD_IN_SOURCE   FALSE
    CONFIGURE_COMMAND ${CONFIGURE_COMMAND}
    TEST_COMMAND      ""
)
